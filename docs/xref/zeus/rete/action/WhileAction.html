34  public class  WhileAction extends ReteAction{

38          */
39      public void executeAction (Action a, Info info) { 
40            while( true ) {
41                   ValueFunction var = ((ValueFunction)a.head).resolve(info.getBindings());
42                   if ( var == null || var.getID() != ValueFunction.BOOL ) {
43                      Core.USER_ERROR("Cannot resolve '" + a.head + "' in action " + a);
44                      return;
45                   }// end if 
46                   if ( ((BoolFn)var).getValue() ) {
47                      for(int i = 0; i < a.items.size(); i++ ) {
48                          AbstractActionFactory actionFact = new AbstractActionFactory(); 
49                          ActionFactory factory  = actionFact.getActionFactory (); 
50                          try {
51                              Action b = (Action)a.items.elementAt(i);
52                             // System.out.println("generating action");
53                              BasicAction action = factory.getAction(b.type);
54                              action.setActuators(this.conflictHandler,context);
55                             // System.out.println("Action was : " + action.toString()); 
56                              action.executeAction (b,info); }
57                              catch (Exception e) { 
58                                  e.printStackTrace(); }
59                         
60                      }// end for
61                   }// end if 
62            }// end while
63         
64      }// end executeAction
