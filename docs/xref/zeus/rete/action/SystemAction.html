42      public void executeAction (Action a, Info info) { 
43                String output = "";
44                ValueFunction var;
45                for(int j = 0; j < a.items.size(); j++ ) {
46                   var = (ValueFunction)a.items.elementAt(j);
47                   var = var.resolve(info.getBindings());
48                   String str = var.toString();
49                   str = Misc.literalToString(str);
50                   output += str;
51                }
52                try {
53                   Runtime runtime = Runtime.getRuntime();
54                   Process process = runtime.exec(output);
55                }
56                catch(IOException e) {
57                   Core.USER_ERROR(e + " while attempting to execute '" + output + "'");
58                }
59                     
60      }
