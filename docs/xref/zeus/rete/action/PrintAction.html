32

36  public class  PrintAction extends ReteAction{
37      
38      /***
39          print to the nominated printwriter 
40          */
41      public void executeAction (Action a, Info info) { 
42                ValueFunction var = ((ValueFunction)a.head).resolve(info.getBindings());
43                if ( var.getID() != ValueFunction.ID ) {
44                   Core.USER_ERROR("Unresolved writer destination '" + var + "' in action " + a);
45                   return;
46                }
47                Hashtable writers = conflictHandler.getWriters();
48                PrintWriter out = (PrintWriter)writers.get(((IdFn)var).getValue());
49                if ( out == null ) {
50                   Core.USER_ERROR("Unknown writer \'" + var + "' in " + a);
51                   return;
52                }
53  
54                String output = new String("");
55                String str = new String(); 
56                for(int j = 0; j < a.items.size(); j++ ) {
57                   var = (ValueFunction)a.items.elementAt(j);
58                   var = var.resolve(info.getBindings());
59                   str = var.toString();
60                   str = Misc.literalToString(str);
61                   output += str;
62                }
63  		         out.print(output);
64           
65   
66      }
