29  package zeus.visualiser.society;
30  
31  import java.util.*;
32  
33  import zeus.util.*;
34  import zeus.concepts.Relationship;
35  
36  public class SocietyModelEntry {
37  
38    protected String  name = null;
39    protected String  icon = null;
40    protected Vector  relations = new Vector();
41  
42    SocietyModelEntry(String name, String icon) {
43       Core.ERROR(name,1,this);
44       this.name = name;
45       this.icon = icon;
46    }
47  
48    String getName()            { return name; }
49    String getIcon()            { return icon; }
50    void   setIcon(String icon) { this.icon = icon; }
51  
52    Relationship[] getRelations() {
53       Relationship[] output = new Relationship[relations.size()];
54       for(int i = 0; i < output.length; i++ )
55          output[i] = (Relationship)relations.elementAt(i);
56       return output;
57    }
58  
59    void addRelations(Vector input) {
60       Relationship relation;
61       for(int i = 0; i < input.size(); i++ ) {
62          relation = (Relationship)input.elementAt(i);
63          if ( !relations.contains(relation) ) 
64             relations.addElement(relation);
65       }
66    }
67  }
