27  /***
28    This exception is thrown when some inconsistency is detected during some
29    ontological operation.
30  
31    @author Giovanni Rimassa - Universita` di Parma
32    @version $Date: 2003/10/09 13:00:37 $ $Revision: 1.1.1.1 $
33  */
34  public class OntologyException extends Exception {
35  
36    /***
37    @serial
38    */
39    private Throwable nested;
40  
41    /***
42      Create a new exception object.
43      @param msg The message for this exception object.
44    */
45    public OntologyException(String msg) {
46      super(msg);
47    }
48  
49    /***
50      Create a new exception object.
51      @param msg The message for this exception object.
52      @param t The exception wrapped by this object.
53    */
54    public OntologyException(String msg, Throwable t) {
55      super(msg);
56      nested = t;
57    }
58  
59      public void printStackTrace() {
60        if (nested != null)
61  	nested.printStackTrace();
62        super.printStackTrace();
63      }
64  }
